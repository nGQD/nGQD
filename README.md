<script>
const canvas = document.getElementById('neural-canvas');
const ctx = canvas.getContext('2d');
let W, H, nodes = [];
const N = 55;

function resize() {
  W = canvas.width  = window.innerWidth;
  H = canvas.height = window.innerHeight;
}

function init() {
  nodes = [];
  for (let i = 0; i < N; i++) {
    nodes.push({
      x:  Math.random() * W,
      y:  Math.random() * H,
      vx: (Math.random() - 0.5) * 0.28,
      vy: (Math.random() - 0.5) * 0.28,
      r:  Math.random() * 1.8 + 0.8,
      phase: Math.random() * Math.PI * 2,
      type: Math.random() < 0.6 ? 'gold' : 'silver'
    });
  }
}

function draw() {
  ctx.clearRect(0, 0, W, H);

  nodes.forEach(n => {
    n.x += n.vx; n.y += n.vy; n.phase += 0.018;
    if (n.x < 0 || n.x > W) n.vx *= -1;
    if (n.y < 0 || n.y > H) n.vy *= -1;
  });

  // Connections
  for (let i = 0; i < N; i++) {
    for (let j = i + 1; j < N; j++) {
      const dx = nodes[i].x - nodes[j].x;
      const dy = nodes[i].y - nodes[j].y;
      const d  = Math.sqrt(dx*dx + dy*dy);
      if (d < 170) {
        const a = (1 - d/170) * 0.28;
        const both = nodes[i].type === 'gold' && nodes[j].type === 'gold';
        ctx.strokeStyle = both
          ? `rgba(255,215,0,${a})`
          : `rgba(200,204,216,${a * 0.7})`;
        ctx.lineWidth = 0.7;
        ctx.beginPath();
        ctx.moveTo(nodes[i].x, nodes[i].y);
        ctx.lineTo(nodes[j].x, nodes[j].y);
        ctx.stroke();
      }
    }
  }

  // Nodes
  nodes.forEach((n, i) => {
    const g = 0.5 + 0.5 * Math.sin(n.phase);
    const isGold = n.type === 'gold';
    const col = isGold ? `rgba(255,215,0,${0.5 + 0.5*g})` : `rgba(200,204,216,${0.35 + 0.4*g})`;

    ctx.beginPath();
    ctx.arc(n.x, n.y, n.r * (0.8 + 0.4*g), 0, Math.PI*2);
    ctx.fillStyle = col;
    ctx.fill();

    // Synapse fire
    if (i % 6 === 0 && Math.sin(n.phase * 2.5) > 0.88) {
      const gr = ctx.createRadialGradient(n.x, n.y, 0, n.x, n.y, 22);
      gr.addColorStop(0, `rgba(255,215,0,${0.3 * g})`);
      gr.addColorStop(1, 'rgba(255,215,0,0)');
      ctx.beginPath();
      ctx.arc(n.x, n.y, 22, 0, Math.PI*2);
      ctx.fillStyle = gr;
      ctx.fill();
    }

    // Hexagonal pulse ring on silver nodes
    if (n.type === 'silver' && i % 8 === 0 && Math.sin(n.phase * 1.8) > 0.9) {
      const sz = 14 * g;
      const cx = n.x, cy = n.y;
      ctx.beginPath();
      for (let a = 0; a < 6; a++) {
        const angle = (a * Math.PI / 3) - Math.PI/6;
        const px = cx + sz * Math.cos(angle);
        const py = cy + sz * Math.sin(angle);
        a === 0 ? ctx.moveTo(px, py) : ctx.lineTo(px, py);
      }
      ctx.closePath();
      ctx.strokeStyle = `rgba(212,216,232,${0.25 * g})`;
      ctx.lineWidth = 0.8;
      ctx.stroke();
    }
  });

  requestAnimationFrame(draw);
}

window.addEventListener('resize', () => { resize(); init(); });
resize(); init(); requestAnimationFrame(draw);
</script>
